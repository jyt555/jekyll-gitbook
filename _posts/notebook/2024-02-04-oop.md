类：

```c++
class Rectangle{
private:
    double w, h;
    double area, perimeter;
public:
    Rectangle(double w, double h) : w(w), h(h) {}
    void calc_area(){
        area = w * h;
    }
    void calc_perimeter(){
        perimeter = 2 * (w + h);
    }
};
int main()
{
    Rectangle arr[] = {{2,3}, {5,5}};
    int n = sizeof(arr) / sizeof(arr[0]);
    
    for(Rectangle& r : arr)
    {
        r.calc_area();
        r.calc_perimeter();
    }
    return 0;
}
```
